! Configuratiebestand voor Switch: Klant-Arcade-SW
!
enable
configure terminal
!
! 1. Hostname instellen
hostname Klant-Arcade-SW
!
! 2. Paswoord encryptie & Security Basics
service password-encryption
no ip domain-lookup
ip domain-name gamecenter.local
!
! 3. Paswoord console verbinding
line console 0
 password cisco
 logging synchronous
 login
 exit
!
! 4. SSH & Remote verbindingen (VTY) configuratie
! Genereer crypto keys (vereist voor SSH):
! crypto key generate rsa modulus 2048
!
line vty 0 15
 password cisco
 transport input ssh
 login
 exit
!
! 5. Paswoord privileged EXEC (Enable)
enable secret class
!
! 6. Banner MotD
banner motd ^Toegang enkel voor bevoegd personeel!^
!
! 7. Management IP Configuratie
interface Vlan1
 description Management Interface
 ip address 192.168.20.3 255.255.255.0
 no shutdown
 exit
!
! 8. Default Gateway
ip default-gateway 192.168.20.1
!
! 9. Interface Configuratie
! Uplink naar Core Switch (Trunk via FastEthernet)
interface FastEthernet0/1
 description Uplink naar Klant-Core-SW
 switchport mode trunk
 no shutdown
 exit
!
! 10. Uitschakelen ongebruikte interfaces
interface range FastEthernet0/2 - 24, GigabitEthernet0/1 - 2
 shutdown
 description Unused/Arcade Port
 exit
!
! Opslaan
end
write memory
!
! Klaar!
