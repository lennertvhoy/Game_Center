! Configuratiebestand voor Switch: Klant-Core-SW
!
enable
configure terminal
!
! 1. Hostname instellen
hostname Klant-Core-SW
!
! 2. Paswoord encryptie & Security Basics
service password-encryption
no ip domain-lookup
ip domain-name gamecenter.local
!
! 3. Paswoord console verbinding
line console 0
 password cisco
 logging synchronous
 login
 exit
!
! 4. SSH & Remote verbindingen (VTY) configuratie
! Genereer crypto keys (vereist voor SSH):
! crypto key generate rsa modulus 2048
!
line vty 0 15
 password cisco
 transport input ssh
 login
 exit
!
! 5. Paswoord privileged EXEC (Enable)
enable secret class
!
! 6. Banner MotD
banner motd ^Toegang enkel voor bevoegd personeel!^
!
! 7. Management IP Configuratie
interface Vlan1
 description Management Interface
 ip address 192.168.20.2 255.255.255.0
 no shutdown
 exit
!
! 8. Default Gateway
ip default-gateway 192.168.20.1
!
! 9. Interface & Trunk Configuratie
! Uplink naar Router (Gigabit)
interface GigabitEthernet0/1
 description Uplink naar GC-Router
 switchport mode access
 switchport access vlan 20
 no shutdown
 exit
!
! Downlinks naar Access Switches (Trunks via FastEthernet)
interface FastEthernet0/1
 description Trunk naar Klant-Arcade-SW
 switchport mode trunk
 no shutdown
 exit
!
interface FastEthernet0/2
 description Trunk naar Klant-Gaming-SW
 switchport mode trunk
 no shutdown
 exit
!
! 10. Uitschakelen ongebruikte interfaces (Security)
interface range FastEthernet0/3 - 24, GigabitEthernet0/2
 shutdown
 description Unused Port
 exit
!
! Opslaan
end
write memory
!
! Klaar!
