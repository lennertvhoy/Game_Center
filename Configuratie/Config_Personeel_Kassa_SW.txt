! Configuratiebestand voor Switch: Personeel-Kassa-SW
!
en
conf t
!
! 1. Hostname instellen
hostname Personeel-Kassa-SW
!
! 2. Paswoord encryptie & Security Basics
service password-encryption
!
! 3. Paswoord console verbinding
line console 0
 password cisco
 login
 exit
!
! 4. Remote verbindingen (VTY)
line vty 0 15
 password cisco
 login
 exit
!
! 5. Paswoord privileged EXEC (Enable)
enable secret class
!
! 6. Banner MotD
banner motd ^Toegang enkel voor bevoegd personeel!^
!
! 7. Management IP Configuratie
int Vlan1
 ip address 192.168.10.3 255.255.255.0
 no shut
 exit
!
! 8. Default Gateway
ip default-gateway 192.168.10.1
!
! 9. Ongebruikte interfaces uitschakelen
int range Fa0/2 - 24, Gi0/1 - 2
 shut
 exit
!
! Opslaan
end
copy run start
!
! Klaar!
